alert( null || 2 || undefined );

**Что выведет alert (ИЛИ)?**
Что выведет код ниже?

alert( alert(1) || 2 || alert(3) );

**Что выведет alert (И)?**
Что выведет код ниже?

alert( 1 && null && 2 );

alert( alert(1) && alert(2) );

Что выведет код ниже?

alert( null || 2 && 3 || 4 );

**Что выведет этот код (||=, &&=)?**
Что выведет код ниже?

let value = NaN;

value &&= 10;
value ||= 20;
value &&= 30;
value ||= 40;

alert(value);
